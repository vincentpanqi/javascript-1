				function reverseString(str) {

				  var a = [];//array
				  for (var i = 0; i < str.length; i++) {
					a[i] = str[i];
					// console.log("a[i]="+a[i]);
				  }
				  a.reverse();
				  // join()
				  str = a.join('');
				  return str;
				}

				reverseString("hello");
