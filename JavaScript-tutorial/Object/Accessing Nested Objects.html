Accessing Nested Objects

		var ourStorage = {
		  "desk": {
		    "drawer": "stapler"
		  },
		  "cabinet": {
		    "top drawer": { 
		      "folder1": "a file",
		      "folder2": "secrets"
		    },
		    "bottom drawer": "soda"
		  }
		};
		ourStorage.cabinet["top drawer"].folder2;  // "secrets"
		ourStorage.desk.drawer; // "stapler"
